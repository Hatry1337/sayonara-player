CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
PROJECT(say_gui_resources)

ADD_SUBDIRECTORY(Icons)

SET(RESOURCES
	Icons.qrc
)

IF(WIN32)
	SET(RESOURCES
		${RESOURCES}
		IconsWindows.qrc
	)
ENDIF()

QT5_ADD_RESOURCES(RESOURCES_RCC
	${RESOURCES}
)



ADD_LIBRARY(${PROJECT_NAME} STATIC ${RESOURCES_RCC})

INSTALL(FILES "standard.css" DESTINATION share/sayonara)
INSTALL(FILES "dark.css" DESTINATION share/sayonara)
